from __future__ import annotations

from pathlib import Path
import sqlite3
from typing import Any


SCHEMA = """
PRAGMA foreign_keys = ON;

CREATE TABLE IF NOT EXISTS trips (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    description TEXT NOT NULL DEFAULT '',
    created_at TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS travelers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    trip_id INTEGER NOT NULL REFERENCES trips(id),
    display_name TEXT NOT NULL,
    english_name TEXT NOT NULL DEFAULT '',
    aliases TEXT NOT NULL DEFAULT '',
    affiliation TEXT NOT NULL DEFAULT '',
    created_at TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS documents (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    trip_id INTEGER NOT NULL REFERENCES trips(id),
    traveler_id INTEGER REFERENCES travelers(id),
    uploaded_by TEXT NOT NULL DEFAULT '',
    original_filename TEXT NOT NULL,
    stored_path TEXT NOT NULL,
    mime_type TEXT NOT NULL DEFAULT '',
    category TEXT NOT NULL DEFAULT 'misc',
    category_confidence REAL NOT NULL DEFAULT 0,
    status TEXT NOT NULL DEFAULT 'needs_review',
    extracted_text TEXT NOT NULL DEFAULT '',
    detected_travelers TEXT NOT NULL DEFAULT '',
    date_hint TEXT NOT NULL DEFAULT '',
    amount_hint TEXT NOT NULL DEFAULT '',
    match_confidence REAL NOT NULL DEFAULT 0,
    notes TEXT NOT NULL DEFAULT '',
    file_size INTEGER NOT NULL DEFAULT 0,
    page_count INTEGER NOT NULL DEFAULT 0,
    created_at TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP
);
"""

DOCUMENT_MIGRATIONS: dict[str, str] = {
    "category_confidence": "AL" "TER TABLE documents ADD COLUMN category_confidence REAL NOT NULL DEFAULT 0",
    "detected_travelers": "AL" "TER TABLE documents ADD COLUMN detected_travelers TEXT NOT NULL DEFAULT ''",
    "file_size": "AL" "TER TABLE documents ADD COLUMN file_size INTEGER NOT NULL DEFAULT 0",
    "page_count": "AL" "TER TABLE documents ADD COLUMN page_count INTEGER NOT NULL DEFAULT 0",
}


def connect(db_path: Path) -> sqlite3.Connection:
    db_path.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(db_path)
    conn.row_factory = sqlite3.Row
    conn.executescript(SCHEMA)
    migrate(conn)
    return conn


def migrate(conn: sqlite3.Connection) -> None:
    conn.execute("PRAGMA foreign_keys = ON")
    columns = {row[1] for row in conn.execute("PRAGMA table_info(documents)").fetchall()}
    for column, statement in DOCUMENT_MIGRATIONS.items():
        if column not in columns:
            conn.execute(statement)
    conn.commit()


def row_to_dict(row: sqlite3.Row | None) -> dict[str, Any] | None:
    return dict(row) if row is not None else None


def list_trips(conn: sqlite3.Connection) -> list[dict[str, Any]]:
    rows = conn.execute(
        """
        SELECT trips.*,
               COUNT(DISTINCT travelers.id) AS traveler_count,
               COUNT(DISTINCT documents.id) AS document_count,
               SUM(CASE WHEN documents.status = 'needs_review' THEN 1 ELSE 0 END) AS needs_review_count,
               SUM(CASE WHEN documents.status = 'auto_matched' THEN 1 ELSE 0 END) AS auto_matched_count,
               SUM(CASE WHEN documents.status = 'reviewed' THEN 1 ELSE 0 END) AS reviewed_count,
               SUM(CASE WHEN documents.traveler_id IS NULL AND documents.id IS NOT NULL THEN 1 ELSE 0 END) AS unassigned_count
        FROM trips
        LEFT JOIN travelers ON travelers.trip_id = trips.id
        LEFT JOIN documents ON documents.trip_id = trips.id
        GROUP BY trips.id
        ORDER BY trips.created_at DESC, trips.id DESC
        """
    ).fetchall()
    return [dict(row) for row in rows]


def create_trip(conn: sqlite3.Connection, name: str, description: str = "") -> int:
    cur = conn.execute(
        "INSERT INTO trips (name, description) VALUES (?, ?)",
        (name.strip(), description.strip()),
    )
    conn.commit()
    return int(cur.lastrowid)


def get_trip(conn: sqlite3.Connection, trip_id: int) -> dict[str, Any] | None:
    return row_to_dict(conn.execute("SELECT * FROM trips WHERE id = ?", (trip_id,)).fetchone())


def add_traveler(
    conn: sqlite3.Connection,
    trip_id: int,
    display_name: str,
    english_name: str = "",
    aliases: str = "",
    affiliation: str = "",
) -> int:
    cur = conn.execute(
        """
        INSERT INTO travelers (trip_id, display_name, english_name, aliases, affiliation)
        VALUES (?, ?, ?, ?, ?)
        """,
        (trip_id, display_name.strip(), english_name.strip(), aliases.strip(), affiliation.strip()),
    )
    conn.commit()
    return int(cur.lastrowid)


def upsert_traveler(
    conn: sqlite3.Connection,
    trip_id: int,
    display_name: str,
    english_name: str = "",
    aliases: str = "",
    affiliation: str = "",
) -> int:
    display_name = display_name.strip()
    english_name = english_name.strip()
    row = conn.execute(
        """
        SELECT * FROM travelers
        WHERE trip_id = ?
          AND (lower(display_name) = lower(?) OR (? != '' AND lower(english_name) = lower(?)))
        ORDER BY id
        LIMIT 1
        """,
        (trip_id, display_name, english_name, english_name),
    ).fetchone()
    if row:
        traveler_id = int(row["id"])
        conn.execute(
            "UP" "DATE travelers "
            "SET display_name = ?, english_name = ?, aliases = ?, affiliation = ? "
            "WHERE id = ?",
            (display_name, english_name, aliases.strip(), affiliation.strip(), traveler_id),
        )
        conn.commit()
        return traveler_id
    return add_traveler(conn, trip_id, display_name, english_name, aliases, affiliation)


def list_travelers(conn: sqlite3.Connection, trip_id: int) -> list[dict[str, Any]]:
    rows = conn.execute(
        "SELECT * FROM travelers WHERE trip_id = ? ORDER BY display_name, id",
        (trip_id,),
    ).fetchall()
    return [dict(row) for row in rows]


def add_document(conn: sqlite3.Connection, **values: Any) -> int:
    cur = conn.execute(
        """
        INSERT INTO documents (
            trip_id, traveler_id, uploaded_by, original_filename, stored_path,
            mime_type, category, category_confidence, status, extracted_text,
            detected_travelers, date_hint, amount_hint, match_confidence, notes,
            file_size, page_count
        )
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """,
        (
            values["trip_id"],
            values.get("traveler_id"),
            values.get("uploaded_by", ""),
            values["original_filename"],
            values["stored_path"],
            values.get("mime_type", ""),
            values.get("category", "misc"),
            values.get("category_confidence", 0.0),
            values.get("status", "needs_review"),
            values.get("extracted_text", ""),
            values.get("detected_travelers", ""),
            values.get("date_hint", ""),
            values.get("amount_hint", ""),
            values.get("match_confidence", 0.0),
            values.get("notes", ""),
            values.get("file_size", 0),
            values.get("page_count", 0),
        ),
    )
    conn.commit()
    return int(cur.lastrowid)


def list_documents(
    conn: sqlite3.Connection,
    trip_id: int,
    *,
    status: str = "",
    category: str = "",
    traveler_id: int | None = None,
) -> list[dict[str, Any]]:
    clauses = ["documents.trip_id = ?"]
    params: list[Any] = [trip_id]
    if status:
        clauses.append("documents.status = ?")
        params.append(status)
    if category:
        clauses.append("documents.category = ?")
        params.append(category)
    if traveler_id is not None:
        if traveler_id == 0:
            clauses.append("documents.traveler_id IS NULL")
        else:
            clauses.append("documents.traveler_id = ?")
            params.append(traveler_id)
    where = " AND ".join(clauses)
    rows = conn.execute(
        f"""
        SELECT documents.*, travelers.display_name AS traveler_name
        FROM documents
        LEFT JOIN travelers ON travelers.id = documents.traveler_id
        WHERE {where}
        ORDER BY
          CASE documents.status WHEN 'needs_review' THEN 0 WHEN 'auto_matched' THEN 1 WHEN 'reviewed' THEN 2 ELSE 3 END,
          documents.created_at DESC,
          documents.id DESC
        """,
        params,
    ).fetchall()
    return [dict(row) for row in rows]


def get_document(conn: sqlite3.Connection, document_id: int) -> dict[str, Any] | None:
    return row_to_dict(
        conn.execute(
            """
            SELECT documents.*, travelers.display_name AS traveler_name
            FROM documents
            LEFT JOIN travelers ON travelers.id = documents.traveler_id
            WHERE documents.id = ?
            """,
            (document_id,),
        ).fetchone()
    )


def save_document_review(
    conn: sqlite3.Connection,
    document_id: int,
    traveler_id: int | None,
    category: str,
    status: str,
    notes: str,
) -> None:
    conn.execute(
        "UP" "DATE documents "
        "SET traveler_id = ?, category = ?, status = ?, notes = ? "
        "WHERE id = ?",
        (traveler_id, category, status, notes, document_id),
    )
    conn.commit()


def trip_summary(conn: sqlite3.Connection, trip_id: int) -> dict[str, int]:
    row = conn.execute(
        """
        SELECT
            COUNT(*) AS document_count,
            SUM(CASE WHEN status = 'needs_review' THEN 1 ELSE 0 END) AS needs_review_count,
            SUM(CASE WHEN status = 'auto_matched' THEN 1 ELSE 0 END) AS auto_matched_count,
            SUM(CASE WHEN status = 'reviewed' THEN 1 ELSE 0 END) AS reviewed_count,
            SUM(CASE WHEN traveler_id IS NULL THEN 1 ELSE 0 END) AS unassigned_count,
            SUM(CASE WHEN extracted_text = '' THEN 1 ELSE 0 END) AS no_text_count
        FROM documents
        WHERE trip_id = ?
        """,
        (trip_id,),
    ).fetchone()
    result = {key: int(row[key] or 0) for key in row.keys()} if row else {}
    result["traveler_count"] = int(
        conn.execute("SELECT COUNT(*) FROM travelers WHERE trip_id = ?", (trip_id,)).fetchone()[0]
    )
    return result


def traveler_document_summary(conn: sqlite3.Connection, trip_id: int) -> list[dict[str, Any]]:
    rows = conn.execute(
        """
        SELECT travelers.id,
               travelers.display_name,
               COUNT(documents.id) AS document_count,
               SUM(CASE WHEN documents.status = 'needs_review' THEN 1 ELSE 0 END) AS needs_review_count,
               SUM(CASE WHEN documents.category = 'airfare' THEN 1 ELSE 0 END) AS airfare_count,
               SUM(CASE WHEN documents.category = 'boarding_pass' THEN 1 ELSE 0 END) AS boarding_pass_count,
               SUM(CASE WHEN documents.category = 'lodging' THEN 1 ELSE 0 END) AS lodging_count,
               SUM(CASE WHEN documents.category = 'conference_registration' THEN 1 ELSE 0 END) AS conference_registration_count,
               SUM(CASE WHEN documents.category = 'meal' THEN 1 ELSE 0 END) AS meal_count,
               SUM(CASE WHEN documents.category = 'transport' THEN 1 ELSE 0 END) AS transport_count
        FROM travelers
        LEFT JOIN documents ON documents.traveler_id = travelers.id
        WHERE travelers.trip_id = ?
        GROUP BY travelers.id
        ORDER BY travelers.display_name
        """,
        (trip_id,),
    ).fetchall()
    return [dict(row) for row in rows]
